print("--- Promedio de edades de 6 estudiantes ---")

suma_edades = 0

for i in range(6):
    edad = int(input(f"Ingrese la edad del estudiante {i + 1}: "))
    suma_edades += edad

promedio = suma_edades / 6

print("\n--- Resultado ---")
print("Suma total de edades:", suma_edades)
print(f"Promedio de edades: {promedio:.2f}")
